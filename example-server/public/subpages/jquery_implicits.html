Improve our Implicit with JQuery

class JQueryUserPrinter extends UserPrinter {

    override def print(user: User): Unit = {
        val userNode = $("<div></div>").text(name(user))
        $("body").append(userNode)
    }

}
def printUser(user: User)(implicit p: UserPrinter) =
    p.print(user)

implicit val userPrinter = new JQueryUserPrinter
printUser(rafaUser)
printUser(fedeUser)
