def getUsers: Future[UserResponse] =
    Ajax.get("http://api.com/user/20") map { request =>
        read[UserResponse](request.responseText)
    }

def getWeather: Future[WeatherResponse] =
    Ajax.get("http://api.com/weather/11130") map { request =>
        read[WeatherResponse](request.responseText)
    }
