Recursively instead of mutability

def drawStuff(implicit ctx: CanvasRenderingContext2D) {
    var x = 0.0
    dom.setInterval(() => {
        x = x + 1
        ctx.fillStyle = "#FF9900"
        ctx.fillRect(x, 100, 30, 30)
    }, 10)
}

def drawStuff(x: Double)(implicit ctx: CanvasRenderingContext2D) {
    dom.setTimeout(() => {
        ctx.fillStyle = "#FF9900"
        ctx.fillRect(x, 100, 30, 30)
        drawStuff(x+1)
    }, 10)
}

drawStuff(0)
